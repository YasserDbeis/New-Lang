func main()
{
    int i = 1;
    
    int three_counter = 1;
    int five_counter = 1;
    int fifteen_counter = 1;
    while (i != 16)
    {
        if (fifteen_counter is 15)
        {
            print(i + ": Fizzbuzz");
            fifteen_counter = 0;
        }
        elsif (three_counter is 3)
        {
            print(i + ": Fizz");
            three_counter = 0;
        }
        elsif (five_counter is 5)
        {
            print(i + ": Buzz");
            five_counter = 0;
        }
        
        three_counter = three_counter + 1;
        five_counter = five_counter + 1;
        fifteen_counter = fifteen_counter + 1;
        i = i + 1;
    }
}
